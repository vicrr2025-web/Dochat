## 二、数据模型（6 表）

```sql
CREATE TABLE express_orders (
    id BIGSERIAL PRIMARY KEY,
    order_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    driver_id VARCHAR(32),
    type VARCHAR(16) NOT NULL,        -- taxi/errand/freight
    vehicle_type VARCHAR(32),         -- 快车/专车/拼车/小面/中面/4.2米...
    origin_address VARCHAR(256),
    origin_lat DECIMAL(10,6),
    origin_lng DECIMAL(10,6),
    dest_address VARCHAR(256),
    dest_lat DECIMAL(10,6),
    dest_lng DECIMAL(10,6),
    cargo_info TEXT,                  -- JSON: {weight, type, ...}
    estimated_price DECIMAL(10,2),
    actual_price DECIMAL(10,2),
    insured BOOLEAN DEFAULT FALSE,
    insurance_fee DECIMAL(10,2),
    status VARCHAR(16) DEFAULT 'pending', -- pending/accepted/pickup/transit/delivered/completed/cancelled/disputed
    rating INT,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE express_drivers (
    id BIGSERIAL PRIMARY KEY,
    driver_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) UNIQUE NOT NULL,
    name VARCHAR(64),
    phone VARCHAR(20),
    id_card VARCHAR(18),
    license_no VARCHAR(32),
    vehicle_type VARCHAR(32),
    vehicle_plate VARCHAR(16),
    vehicle_info TEXT,                -- JSON
    auth_status VARCHAR(16) DEFAULT 'pending', -- pending/approved/rejected
    credit_score INT DEFAULT 100,
    status VARCHAR(16) DEFAULT 'offline',  -- online/offline/busy
    current_lat DECIMAL(10,6),
    current_lng DECIMAL(10,6),
    total_income DECIMAL(10,2) DEFAULT 0,
    balance DECIMAL(10,2) DEFAULT 0,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE express_locations (
    id BIGSERIAL PRIMARY KEY,
    record_id VARCHAR(32) UNIQUE NOT NULL,
    order_id VARCHAR(32) NOT NULL,
    lat DECIMAL(10,6),
    lng DECIMAL(10,6),
    recorded_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE express_disputes (
    id BIGSERIAL PRIMARY KEY,
    dispute_id VARCHAR(32) UNIQUE NOT NULL,
    order_id VARCHAR(32) NOT NULL,
    initiator_id VARCHAR(32) NOT NULL,
    reason VARCHAR(512),
    evidence TEXT,                    -- JSON array of URLs
    jury_votes_for INT DEFAULT 0,
    jury_votes_against INT DEFAULT 0,
    total_jurors INT DEFAULT 17,
    verdict VARCHAR(16),              -- pending/for_user/for_driver
    status VARCHAR(16) DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE express_insurances (
    id BIGSERIAL PRIMARY KEY,
    insurance_id VARCHAR(32) UNIQUE NOT NULL,
    order_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    type VARCHAR(32) DEFAULT 'basic',  -- basic/standard/premium
    fee DECIMAL(10,2),
    coverage DECIMAL(10,2),
    status VARCHAR(16) DEFAULT 'active',
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE express_withdrawals (
    id BIGSERIAL PRIMARY KEY,
    withdrawal_id VARCHAR(32) UNIQUE NOT NULL,
    driver_id VARCHAR(32) NOT NULL,
    amount DECIMAL(10,2),
    bank_account VARCHAR(64),
    status VARCHAR(16) DEFAULT 'pending',  -- pending/success/failed
    created_at TIMESTAMP DEFAULT NOW()
);
```
