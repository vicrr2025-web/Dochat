## 二、数据模型（5 表）

```sql
CREATE TABLE mail_accounts (
    id BIGSERIAL PRIMARY KEY,
    account_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    email VARCHAR(128) NOT NULL,
    password VARCHAR(256) NOT NULL,     -- 加密存储
    provider VARCHAR(32) DEFAULT 'custom',  -- qq/163/gmail/outlook/custom
    imap_host VARCHAR(128),
    imap_port INT DEFAULT 993,
    smtp_host VARCHAR(128),
    smtp_port INT DEFAULT 465,
    display_name VARCHAR(64),
    is_default BOOLEAN DEFAULT FALSE,
    status VARCHAR(16) DEFAULT 'active',
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE mail_messages (
    id BIGSERIAL PRIMARY KEY,
    message_id VARCHAR(32) UNIQUE NOT NULL,
    account_id VARCHAR(32) NOT NULL,
    folder VARCHAR(32) DEFAULT 'inbox',  -- inbox/sent/drafts/trash/custom_folder
    message_uid VARCHAR(128),            -- IMAP UID
    sender VARCHAR(256),
    recipients TEXT,                     -- JSON array
    subject VARCHAR(512),
    body TEXT,
    is_read BOOLEAN DEFAULT FALSE,
    is_starred BOOLEAN DEFAULT FALSE,
    has_attachments BOOLEAN DEFAULT FALSE,
    attachments TEXT,                    -- JSON array [{name, size, url}]
    received_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE mail_folders (
    id BIGSERIAL PRIMARY KEY,
    folder_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    name VARCHAR(64) NOT NULL,
    icon VARCHAR(32),
    sort_order INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE mail_filters (
    id BIGSERIAL PRIMARY KEY,
    filter_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    type VARCHAR(16) NOT NULL,           -- blacklist/whitelist
    address_pattern VARCHAR(256) NOT NULL,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE mail_reads (
    id BIGSERIAL PRIMARY KEY,
    read_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    article_id VARCHAR(32) NOT NULL,
    title VARCHAR(256),
    source VARCHAR(64),
    url VARCHAR(512),
    is_favorited BOOLEAN DEFAULT FALSE,
    read_at TIMESTAMP DEFAULT NOW()
);
```
