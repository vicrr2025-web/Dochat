## 二、数据模型（3 新表）

```sql
CREATE TABLE mall_products (
    id BIGSERIAL PRIMARY KEY,
    product_id VARCHAR(32) NOT NULL UNIQUE,
    seller_id VARCHAR(32) NOT NULL,
    category VARCHAR(16) NOT NULL,   -- farm/factory/idle
    title VARCHAR(256) NOT NULL,
    description TEXT,
    price DECIMAL(10,2) NOT NULL,
    images TEXT,                      -- JSON array of URLs
    stock INT DEFAULT 1,
    status VARCHAR(16) DEFAULT 'active', -- active/sold/off
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE mall_cart_items (
    id BIGSERIAL PRIMARY KEY,
    user_id VARCHAR(32) NOT NULL,
    product_id VARCHAR(32) NOT NULL,
    quantity INT DEFAULT 1,
    created_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(user_id, product_id)
);

CREATE TABLE mall_orders (
    id BIGSERIAL PRIMARY KEY,
    order_id VARCHAR(32) NOT NULL UNIQUE,
    buyer_id VARCHAR(32) NOT NULL,
    seller_id VARCHAR(32) NOT NULL,
    product_id VARCHAR(32) NOT NULL,
    amount DECIMAL(10,2) NOT NULL,
    quantity INT DEFAULT 1,
    status VARCHAR(16) DEFAULT 'pending', -- pending/paid/shipped/received/completed/cancelled
    session_id VARCHAR(32),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);
```
