## 二、数据模型（5 表）

```sql
CREATE TABLE houses (
    id BIGSERIAL PRIMARY KEY,
    house_id VARCHAR(32) UNIQUE NOT NULL,
    publisher_id VARCHAR(32) NOT NULL,
    type VARCHAR(16) NOT NULL,  -- new/second/rent/commercial
    sub_type VARCHAR(32),       -- 整租/合租/商铺/写字楼/厂房/仓库/土地/车位
    title VARCHAR(256),
    description TEXT,
    price DECIMAL(12,2),
    price_unit VARCHAR(16),     -- 万/元/月/万/月
    area DECIMAL(8,2),          -- 平方米
    layout VARCHAR(32),         -- 户型 3室2厅
    floor_info VARCHAR(64),     -- 楼层信息
    direction VARCHAR(16),      -- 朝向
    decoration VARCHAR(16),     -- 装修
    community_name VARCHAR(128),
    address VARCHAR(256),
    longitude DECIMAL(10,6),
    latitude DECIMAL(10,6),
    images TEXT,                -- JSON array
    tags TEXT,                  -- JSON array
    status VARCHAR(16) DEFAULT 'pending',  -- pending/approved/rejected/sold
    view_count INT DEFAULT 0,
    favorite_count INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE house_appointments (
    id BIGSERIAL PRIMARY KEY,
    appointment_id VARCHAR(32) UNIQUE NOT NULL,
    house_id VARCHAR(32) NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    appointment_time TIMESTAMP NOT NULL,
    contact_name VARCHAR(64),
    contact_phone VARCHAR(20),
    remark VARCHAR(256),
    type VARCHAR(16) DEFAULT 'viewing',  -- viewing/bus
    status VARCHAR(16) DEFAULT 'pending', -- pending/confirmed/completed/cancelled
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE house_favorites (
    id BIGSERIAL PRIMARY KEY,
    user_id VARCHAR(32) NOT NULL,
    house_id VARCHAR(32) NOT NULL,
    created_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(user_id, house_id)
);

CREATE TABLE house_evaluations (
    id BIGSERIAL PRIMARY KEY,
    evaluation_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    type VARCHAR(16) NOT NULL,   -- valuation/renovation
    house_info TEXT,             -- JSON
    result TEXT,                 -- JSON
    status VARCHAR(16) DEFAULT 'completed',
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE house_vips (
    id BIGSERIAL PRIMARY KEY,
    vip_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    level INT DEFAULT 1,
    expires_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);
```
