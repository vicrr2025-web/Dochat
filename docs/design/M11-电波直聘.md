## 二、数据模型（7 表）

```sql
CREATE TABLE job_resumes (
    id BIGSERIAL PRIMARY KEY,
    resume_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) UNIQUE NOT NULL,
    name VARCHAR(64),
    avatar VARCHAR(256),
    phone VARCHAR(20),
    email VARCHAR(128),
    education TEXT,              -- JSON: [{school, major, degree, startDate, endDate}]
    experience TEXT,             -- JSON: [{company, position, startDate, endDate, description}]
    skills TEXT,                 -- JSON array
    intention TEXT,              -- JSON: {industry, position, city, salaryMin, salaryMax}
    privacy VARCHAR(16) DEFAULT 'public',  -- public/friends/private
    exposure INT DEFAULT 0,
    attachment VARCHAR(256),     -- 附件简历 URL
    status VARCHAR(16) DEFAULT 'active',
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE job_positions (
    id BIGSERIAL PRIMARY KEY,
    position_id VARCHAR(32) UNIQUE NOT NULL,
    company_id VARCHAR(32) NOT NULL,
    company_name VARCHAR(128),
    title VARCHAR(128) NOT NULL,
    industry VARCHAR(64),
    city VARCHAR(32),
    salary_min INT,
    salary_max INT,
    experience_required VARCHAR(32),
    education_required VARCHAR(32),
    description TEXT,
    tags TEXT,                   -- JSON array
    status VARCHAR(16) DEFAULT 'active',  -- active/paused/closed
    view_count INT DEFAULT 0,
    apply_count INT DEFAULT 0,
    is_boosted BOOLEAN DEFAULT FALSE,
    boost_expires_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE job_applications (
    id BIGSERIAL PRIMARY KEY,
    application_id VARCHAR(32) UNIQUE NOT NULL,
    position_id VARCHAR(32) NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    resume_id VARCHAR(32),
    status VARCHAR(16) DEFAULT 'applied',  -- applied/viewed/communicating/interview/offer/rejected
    greeting TEXT,               -- 打招呼内容
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(position_id, user_id)
);

CREATE TABLE job_interviews (
    id BIGSERIAL PRIMARY KEY,
    interview_id VARCHAR(32) UNIQUE NOT NULL,
    application_id VARCHAR(32) NOT NULL,
    company_user_id VARCHAR(32) NOT NULL,
    candidate_user_id VARCHAR(32) NOT NULL,
    interview_time TIMESTAMP,
    location VARCHAR(256),
    type VARCHAR(16) DEFAULT 'offline',  -- offline/online
    status VARCHAR(16) DEFAULT 'pending', -- pending/confirmed/sign_in/completed/rated
    remark VARCHAR(512),
    candidate_rating INT,
    company_rating INT,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE job_favorites (
    id BIGSERIAL PRIMARY KEY,
    user_id VARCHAR(32) NOT NULL,
    resume_id VARCHAR(32) NOT NULL,
    folder VARCHAR(64) DEFAULT '默认',
    created_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(user_id, resume_id)
);

CREATE TABLE job_companies (
    id BIGSERIAL PRIMARY KEY,
    company_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) UNIQUE NOT NULL,
    name VARCHAR(128) NOT NULL,
    logo VARCHAR(256),
    industry VARCHAR(64),
    scale VARCHAR(32),
    address VARCHAR(256),
    license_url VARCHAR(256),
    auth_status VARCHAR(16) DEFAULT 'pending',  -- pending/approved/rejected
    description TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE job_vips (
    id BIGSERIAL PRIMARY KEY,
    vip_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    role VARCHAR(16) DEFAULT 'candidate',  -- candidate/recruiter
    level INT DEFAULT 1,
    expires_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);
```
