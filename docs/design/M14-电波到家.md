## 二、数据模型（7 表）

```sql
CREATE TABLE home_services (
    id BIGSERIAL PRIMARY KEY,
    service_id VARCHAR(32) UNIQUE NOT NULL,
    category_id VARCHAR(32),
    name VARCHAR(128) NOT NULL,
    description TEXT,
    price_type VARCHAR(16) DEFAULT 'fixed',  -- fixed/quote/calculator
    base_price DECIMAL(10,2),
    unit VARCHAR(32),
    images TEXT,
    tags TEXT,
    status VARCHAR(16) DEFAULT 'active',
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE home_categories (
    id BIGSERIAL PRIMARY KEY,
    category_id VARCHAR(32) UNIQUE NOT NULL,
    parent_id VARCHAR(32),
    name VARCHAR(64) NOT NULL,
    level INT DEFAULT 1,        -- 1/2/3
    icon VARCHAR(64),
    sort_order INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE home_orders (
    id BIGSERIAL PRIMARY KEY,
    order_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    worker_id VARCHAR(32),
    service_id VARCHAR(32) NOT NULL,
    address VARCHAR(256),
    appointment_time TIMESTAMP,
    price DECIMAL(10,2),
    status VARCHAR(16) DEFAULT 'pending',  -- pending/accepted/dispatched/serving/waiting_verify/completed/cancelled/disputed/after_sale
    rating INT,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE home_workers (
    id BIGSERIAL PRIMARY KEY,
    worker_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) UNIQUE NOT NULL,
    name VARCHAR(64),
    phone VARCHAR(20),
    id_card VARCHAR(18),
    skills TEXT,                   -- JSON array of category_ids
    certificates TEXT,             -- JSON array of certificate URLs
    deposit DECIMAL(10,2) DEFAULT 0,
    auth_status VARCHAR(16) DEFAULT 'pending',
    credit_score INT DEFAULT 100,
    status VARCHAR(16) DEFAULT 'offline',
    total_income DECIMAL(10,2) DEFAULT 0,
    balance DECIMAL(10,2) DEFAULT 0,
    completed_orders INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE home_favorites (
    id BIGSERIAL PRIMARY KEY,
    user_id VARCHAR(32) NOT NULL,
    worker_id VARCHAR(32) NOT NULL,
    created_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(user_id, worker_id)
);

CREATE TABLE home_disputes (
    id BIGSERIAL PRIMARY KEY,
    dispute_id VARCHAR(32) UNIQUE NOT NULL,
    order_id VARCHAR(32) NOT NULL,
    initiator_id VARCHAR(32) NOT NULL,
    reason VARCHAR(512),
    evidence TEXT,
    jury_votes_for INT DEFAULT 0,
    jury_votes_against INT DEFAULT 0,
    total_jurors INT DEFAULT 17,
    verdict VARCHAR(16),
    status VARCHAR(16) DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE home_trainings (
    id BIGSERIAL PRIMARY KEY,
    training_id VARCHAR(32) UNIQUE NOT NULL,
    title VARCHAR(128),
    category VARCHAR(32),       -- skill/rule/safety
    content TEXT,
    duration INT,               -- minutes
    created_at TIMESTAMP DEFAULT NOW()
);
```
