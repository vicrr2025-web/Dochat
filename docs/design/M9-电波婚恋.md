## 二、数据模型（5 表）

```sql
CREATE TABLE dating_profiles (
    id BIGSERIAL PRIMARY KEY,
    profile_id VARCHAR(32) UNIQUE NOT NULL,  -- 独立UUID16，与user_id一一对应
    user_id VARCHAR(32) UNIQUE NOT NULL,
    gender VARCHAR(8) NOT NULL,
    birthday DATE,
    height INT,
    education VARCHAR(32),
    income VARCHAR(32),
    marital_status VARCHAR(16),
    avatar VARCHAR(256),
    photos TEXT,              -- JSON array
    tags TEXT,                -- JSON array
    about_me TEXT,
    real_verified BOOLEAN DEFAULT FALSE,
    work_verified BOOLEAN DEFAULT FALSE,
    edu_verified BOOLEAN DEFAULT FALSE,
    charm_value INT DEFAULT 0,
    love_coin INT DEFAULT 0,
    vip_level INT DEFAULT 0,
    vip_expires_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE dating_likes (
    id BIGSERIAL PRIMARY KEY,
    from_user_id VARCHAR(32) NOT NULL,
    to_user_id VARCHAR(32) NOT NULL,
    like_type VARCHAR(16) DEFAULT 'normal',  -- normal/super
    is_matched BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(from_user_id, to_user_id)
);

CREATE TABLE dating_notes (
    id BIGSERIAL PRIMARY KEY,
    note_id VARCHAR(32) UNIQUE NOT NULL,
    from_user_id VARCHAR(32) NOT NULL,
    to_user_id VARCHAR(32) NOT NULL,
    content TEXT,
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE dating_feeds (
    id BIGSERIAL PRIMARY KEY,
    feed_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    content TEXT,
    images TEXT,
    like_count INT DEFAULT 0,
    comment_count INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE dating_lives (
    id BIGSERIAL PRIMARY KEY,
    live_id VARCHAR(32) UNIQUE NOT NULL,
    user_id VARCHAR(32) NOT NULL,
    status VARCHAR(16) DEFAULT 'live',
    viewer_count INT DEFAULT 0,
    gift_value INT DEFAULT 0,
    started_at TIMESTAMP DEFAULT NOW(),
    ended_at TIMESTAMP
);
```
